PROGRAM a1;
    VAR x, y, z, m, n:INTEGER; f:REAL
    BEGIN
        x:=+1,2    // x=1+2
        y:=-*x,3,2  // y=x*3-2
        z:=-/y,2,5  // z=y/2-5
        IF x < z THEN 
            x:=+1,x 
        ELSE 
            z:=-1,z
        WHILE x <= y DO 
            x:=+1,x 
        BEGIN
            f:=1
            m:=2
            n:=3
        END 
    END.
