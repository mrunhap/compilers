program a1;
    var x, y, z, m, n:integer; f:real
    begin
        x:=+1,2    // x=1+2
        y:=-*x,3,2  // y=x*3-2
        z:=-/y,2,5  // z=y/2-5
        if x < z then 
            x:=+1,x 
        else 
            z:=-1,z
        while x <= y do 
            x:=+1,x 
        begin
            f:=1
            m:=2
            n:=3
        end 
    end.
